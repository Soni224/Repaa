  5699→function setupSwipeDelete() {
  5700→  const items = document.querySelectorAll('.saved-item');
  5701→  items.forEach(item => {
  5702→    let startX = 0, currentX = 0, isDragging = false;
  5703→    const card = item.querySelector('.saved-card');

  5706→      startX = e.touches[0].clientX;
  5707→      isDragging = true;
  5708→      item.classList.add('swiping');
  5709→    }, { passive: true });
  5710→
  5711→    card.addEventListener('touchmove', e => {
  5712→      if (!isDragging) return;
  5713→      currentX = e.touches[0].clientX - startX;
  5714→      if (currentX < 0) {
  5715→        const clamped = Math.max(currentX, -80);
  5716→        card.style.transform = 'translateX(' + clamped + 'px)';
  5717→      }
  5718→    }, { passive: true });
  5719→
  5720→    card.addEventListener('touchend', () => {
  5721→      isDragging = false;
  5722→      item.classList.remove('swiping');
  5723→      if (currentX < -40) {
  5724→        // Reveal delete
  5725→        card.style.transform = 'translateX(-80px)';
  5726→      } else {
  5727→        card.style.transform = 'translateX(0)';
  5728→      }
  5729→      currentX = 0;
  5730→    });
  5731→
  5732→    // Click outside to reset
  5733→    document.addEventListener('click', e => {
  5734→      if (!item.contains(e.target)) {
  5735→        card.style.transform = 'translateX(0)';
  5736→      }
  5737→    });
  5738→  });
  5739→}
