  6168→function setupSwipeDelete() {
  6169→  const items = document.querySelectorAll('.saved-item');
  6170→  items.forEach(item => {
  6171→    let startX = 0, currentX = 0, isDragging = false;
  6172→    const card = item.querySelector('.saved-card');
  6173→
  6174→    card.addEventListener('touchstart', e => {
  6175→      startX = e.touches[0].clientX;
  6176→      isDragging = true;
  6177→      item.classList.add('swiping');
  6178→    }, { passive: true });
  6179→
  6180→    card.addEventListener('touchmove', e => {
  6181→      if (!isDragging) return;
  6182→      currentX = e.touches[0].clientX - startX;
  6183→      if (currentX < 0) {
  6184→        const clamped = Math.max(currentX, -80);
  6185→        card.style.transform = 'translateX(' + clamped + 'px)';
  6186→      }
  6187→    }, { passive: true });
  6188→
  6189→    card.addEventListener('touchend', () => {
  6190→      isDragging = false;
  6191→      item.classList.remove('swiping');
  6192→      if (currentX < -40) {
  6193→        // Reveal delete
  6194→        card.style.transform = 'translateX(-80px)';
  6195→      } else {
  6196→        card.style.transform = 'translateX(0)';
  6197→      }
  6198→      currentX = 0;
  6199→    });
  6200→
  6201→    // Click outside to reset
  6202→    document.addEventListener('click', e => {
  6203→      if (!item.contains(e.target)) {
  6204→        card.style.transform = 'translateX(0)';
  6205→      }
  6206→    });
  6207→  });
  6208→}
